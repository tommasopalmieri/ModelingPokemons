%%%%%%%%%Fill a Matrix with Dots%%%%%%%%%%%%%%%%%%
\newcommand{\dottedcolumn}[3]{%
	\settowidth{\dimen0}{$#1$}
	\settowidth{\dimen2}{$#2$}
	\ifdim\dimen2>\dimen0 \dimen0=\dimen2 \fi
	\begin{pmatrix}\,
		\vcenter{
			\kern.6ex
			\vbox to \dimexpr#1\normalbaselineskip-1.2ex{
				\hbox{$#2$}
				\kern3pt
				\xleaders\vbox{\hbox to \dimen0{\hss.\hss}\vskip4pt}\vfill
				\kern1pt
				\hbox{$#3$}
			}\kern.6ex}\,
	\end{pmatrix}
}
